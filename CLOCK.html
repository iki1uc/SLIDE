import { FINALCLOU } from "./CLOU.js"; import { CLOCK } from "./CLOCK.js"; import { SLIDE } from "./SLIDE.js"; import { RESPO } from "./RESPO.js"; export function pipeline4(step, calc) { const tick = CLOCK(step); const clou = FINALCLOU(calc); const slide = SLIDE(clou); const respo = RESPO(slide.shift); return { step, tick, clou, slide, respo }; }
